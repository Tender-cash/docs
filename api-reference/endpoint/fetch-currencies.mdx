title: 'Fetch All Currencies'
description: 'Retrieve all supported currencies with their chain mappings'
openapi: 'GET /v1/api/system/currency'
---

</ParamField>

<ParamField header="x-timestamp" type="string" required>

  **Example:** `"2025-03-15T09:45:53.000Z"`
</ParamField>

<ParamField header="x-request-id" type="string" required>
  Unique identifier for the request (UUID v4)

<ParamField header="x-access-id" type="string" required>

  **Example:** `"your-access-id-here"`

<ResponseField name="status" type="string" required>

  **Example:** `"success"`
</ResponseField>

<ResponseField name="message" type="string" required>

  **Example:** `"success"`
</ResponseField>

<ResponseField name="data" type="array" required>

    <ResponseField name="name" type="string">

      **Example:** `"USDC"`
    </ResponseField>
    
    <ResponseField name="icon" type="string">
      URL to the currency's icon image
      
      **Example:** `"https://secureapi.tender.cash/icons/usdc.png"`
    </ResponseField>
    
    <ResponseField name="isContract" type="boolean">
      Whether this is a smart contract token
      
      **Example:** `true`
    </ResponseField>
    
    <ResponseField name="chains" type="array">
      Array of chains where this currency is available
      
      **Example:** `["avalanche", "ethereum", "polygon", "solana"]`
    </ResponseField>
    
    <ResponseField name="priceTag" type="string">
      Price identifier for market data

    </ResponseField>
    
    <ResponseField name="fee" type="object">
